Uint16ArrayResource class 

class Uint16ArrayResource extends ResourceBase {
 Uint16Array _array;

 Uint16ArrayResource(String url, ResourceManager rm) : super(url, rm) {
 }

 Uint16Array get array => _array;
 set array(Uint16Array array) {
   _array = array;
   _fireUpdated();
 }

 void update(Map state) {
   dynamic o;
   o = state['array'];
   if (o != null && o is Uint16Array) {
     _array = o;
     _fireUpdated();
   }
   o = state['list'];
   if (o != null && o is List<num>) {
     _array = new Uint16Array.fromList(o);
     _fireUpdated();
   }
 }

 void load(_ResourceLoaderResult result) {
   _fireUpdated();
   result.completer.complete(result.handle);
 }

 void unload() {
   _fireUnloaded();
 }
}
